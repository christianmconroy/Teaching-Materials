# Convert Parquet to JSON# Load in packagesimport pandas as pdimport json# Import synthetic file as Pandas DFdf2=pd.read_csv(r'test.csv')# Convert to Parquet and Exportdf2.to_parquet(r'df.parquet.gzip',              compression='gzip',index=False)# Import Parquet back in as Pandas DFdf3=pd.read_parquet(r'df.parquet.gzip')# Sort filenames by distance (Low to High)df3=df3.sort_values(by='distance')# Reset index based on sorted values df3=df3.reset_index(drop=True)# Convert to JSON and export df3.to_json(r'json_dummy.json',orient='table')# Test load back in with open(r'json_dummy.json') as file:    d=json.load(file)    print(d)
